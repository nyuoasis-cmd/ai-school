export function calculateScore(categories, checkStates) {
  let totalItems = 0;
  let checked = 0;
  let flagged = 0;
  let unchecked = 0;
  const flaggedItems = [];

  categories.forEach((cat, catIndex) => {
    cat.items.forEach((item, itemIndex) => {
      totalItems++;
      const key = `${catIndex}-${itemIndex}`;
      const state = checkStates[key];

      if (state === 'checked') {
        checked++;
      } else if (state === 'flagged') {
        flagged++;
        flaggedItems.push({ text: item, cat: cat.name });
      } else {
        unchecked++;
      }
    });
  });

  // ì ìˆ˜ ê³µì‹
  const baseScore = Math.round((checked / totalItems) * 100);
  const flagPenalty = Math.round((flagged / totalItems) * 40);
  const uncheckPenalty = Math.round((unchecked / totalItems) * 10);
  const score = Math.max(0, Math.min(100, baseScore - flagPenalty - uncheckPenalty));

  return {
    score,
    checked,
    flagged,
    unchecked,
    totalItems,
    flaggedItems,
  };
}
